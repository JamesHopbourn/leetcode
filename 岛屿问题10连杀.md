```java
class Solution {
    public int numIslands(char[][] grid) {
        int count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1') {
                    bfs(grid, i, j);
                    count++;
                }
            }
        }
        return count;
    }

   public void bfs(char[][] grid, int i, int j) {
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{i, j});

        while (!queue.isEmpty()) {
            int[] current = queue.poll();
            int x = current[0];
            int y = current[1];

            boolean inArea = x >= 0 && y >= 0 && x < grid.length && y < grid[0].length;
            if (inArea && grid[x][y] == '1') {
                grid[x][y] = '0';
                queue.offer(new int[]{x + 1, y});
                queue.offer(new int[]{x - 1, y});
                queue.offer(new int[]{x, y + 1});
                queue.offer(new int[]{x, y - 1});
            }
        }
    }
}
```

```java
class Solution {
    public int closedIsland(int[][] grid) {
        int r = grid.length, c = grid[0].length;
        for(int j = 0; j < c; j++) {
            //把靠上边的岛屿淹没
            dfs(grid, 0, j);
            //把靠下边的岛屿淹没
            dfs(grid, r-1, j);
        }
        for(int i = 0; i < r; i++) {
            //把靠左边的岛屿淹没
            dfs(grid, i, 0);
            //把靠右边的岛屿淹没
            dfs(grid, i, c-1);
        }
        int count = 0;
        for(int i = 0; i < r; i++) {
            for(int j = 0; j < c; j++) {
                if(grid[i][j] == 0) {
                    count++;
                    dfs(grid, i, j);
                }
            }
        }
        return count;
    }

    public void dfs(int[][] grid, int i, int j) {
        boolean inArea = i >= 0 && j >= 0 && i < grid.length && j < grid[0].length;
        if(!inArea || grid[i][j] == 1) {
            return;
        }
        //将(i,j)变成海水
        grid[i][j] = 1;
        //淹没上下左右的陆地
        dfs(grid, i+1, j);
        dfs(grid, i, j+1);
        dfs(grid, i-1, j);
        dfs(grid, i, j-1);
    }
}
```

```java
public class Solution {
    // 标记是否为子岛屿的变量
    int flag;

    // 主函数，计算grid2中的子岛屿数量
    public int countSubIslands(int[][] grid1, int[][] grid2) {
        int count = 0;
        for (int r = 0; r < grid2.length; r++) {
            for (int c = 0; c < grid2[r].length; c++) {
                // 如果grid2中当前位置是陆地且尚未访问过
                if (grid2[r][c] == 1) {
                    // 初始化标记为1
                    flag = 1;
                    // 进行深度优先搜索
                    dfs(r, c, grid1, grid2);
                    // 如果flag为1，说明当前grid2中的岛屿是grid1的子岛屿
                    // 则将count增加1
                    count += flag;
                }
            }
        }
        return count;
    }

    // 深度优先搜索函数
    private void dfs(int x, int y, int[][] grid1, int[][] grid2) {
        // 边界条件判断
        boolean inArea = x >= 0 && x < grid2.length && y >= 0 && y < grid2[0].length;
        if (!inArea || grid2[x][y] == 0) return;

        // 如果grid1中对应位置是水域，说明不是子岛屿，将flag置为0
        if (grid1[x][y] == 0) {
            flag = 0;
            return;
        }

        // 将当前位置标记为已访问
        grid2[x][y] = 0;
        
        dfs(x + 1, y, grid1, grid2);
        dfs(x - 1, y, grid1, grid2);
        dfs(x, y + 1, grid1, grid2);
        dfs(x, y - 1, grid1, grid2);
    }
}
```

```java
class Solution {
    public boolean exist(char[][] board, String word) {
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (board[i][j] == word.charAt(0)) {
                    // 之所以要在这里再做一个判断，是因为情况可能不止一种
                    if (dfs(board, i, j, word, 0)) {
                        return true;
                    }
                }
            }
        }
        return false;
    }

    private boolean dfs(char[][] board, int i, int j, String word, int index) {
        if (index == word.length()) return true;

        boolean inArea = i >= 0 && i < board.length && j >= 0 && j < board[0].length;
        if (!inArea || board[i][j] != word.charAt(index)) return false;

        char temp = board[i][j];
        board[i][j] = '#';
        boolean exist = dfs(board, i + 1, j, word, index + 1) ||
                        dfs(board, i - 1, j, word, index + 1) ||
                        dfs(board, i, j + 1, word, index + 1) ||
                        dfs(board, i, j - 1, word, index + 1);
        board[i][j] = temp;
        return exist;
    }
}
```
