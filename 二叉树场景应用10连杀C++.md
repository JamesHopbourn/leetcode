```c++
class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        if (root == nullptr) {
            return nullptr;
        }
        TreeNode* node = root->left;
        root->left = root->right;
        root->right = node;
        invertTree(root->left);
        invertTree(root->right);
        return root;
    }
};
```

#### 101. 对称二叉树
```c++
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        queue<TreeNode*> queue;
        queue.push(root);
        queue.push(root);

        while (!queue.empty()){
            TreeNode* left = queue.front();
            queue.pop();
            TreeNode* right = queue.front();
            queue.pop();
            if (left == nullptr && right == nullptr) {
                continue;
            }
            if (left == nullptr || right == nullptr) {
                return false;
            }
            if (left->val != right->val){
                return false;
            }
            queue.push(left->left);
            queue.push(right->right);
            queue.push(left->right);
            queue.push(right->left);
        }
        return true;
    }
};
```

#### 222. 完全二叉树的节点个数
```c++
class Solution {
public:
    int countNodes(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }

        queue<TreeNode*> queue;
        queue.push(root);

        int count = 0;
        while (!queue.empty()) {
            int size = queue.size();
            while (size-- > 0) {
                TreeNode* node = queue.front();
                queue.pop();

                count++;

                if (node->left != nullptr) {
                    queue.push(node->left);
                }

                if (node->right != nullptr) {
                    queue.push(node->right);
                }
            }
        }

        return count;
    }
};
```

#### 257. 二叉树的所有路径
```c++
class Solution {
public:
    vector<string> result;   // List<String> -> vector<string>
    string builder;          // StringBuilder -> string

    vector<string> binaryTreePaths(TreeNode* root) {
        dfs(root, builder);
        return result;
    }

    void dfs(TreeNode* root, string builder) {
        builder += to_string(root->val);
        if (root->left == nullptr && root->right == nullptr) {
            // C++ 直接 push_back(builder)
            result.push_back(builder);
            return;
        }
        if (root->left != nullptr) {
            // C++ 直接 builder + "->"
            dfs(root->left, builder + "->");
        }
        if (root->right != nullptr) {
            dfs(root->right, builder + "->");
        }
    }
};
```

#### 404. 左叶子之和
```c++
class Solution {
public:
    int sumOfLeftLeaves(TreeNode* root) {
        queue<TreeNode*> queue; // C++语法
        queue.push(root);
        int sum = 0;

        while (!queue.empty()) { // isempty() 改成 empty()
            int size = queue.size();
            while (size-->0) {
                TreeNode* node = queue.front(); // 不能直接 pop()，先 front() 获取再 pop()
                queue.pop();

                if (node->left != nullptr) {
                    // boolean 改成 bool
                    bool isLeaf = node->left->left == nullptr && node->left->right == nullptr;
                    if (isLeaf) {
                        sum += node->left->val;
                    } else {
                        // C++ 不是 add() 改成 push()
                        queue.push(node->left);
                    }
                }
                if (node->right != nullptr) {
                    // C++ 不是 add() 改成 push()
                    queue.push(node->right);
                }
            }
        }
        return sum;
    }
};
```

```c++
        int result = 0;

        while (!queue.empty()) {
            int size = queue.size();
            while (size-- > 0) {
                TreeNode* node = queue.front();
                queue.pop();

                result = node->val;

                if (node->right != nullptr) {
                    queue.push(node->right);
                }

                if (node->left != nullptr) {
                    queue.push(node->left);
                }
            }
        }

        return result;
    }
};
```

#### 112. 路径总和
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        if (root == nullptr){
            return false;
        }
        queue<TreeNode*> queue;
        queue.push(root);
        while (!queue.empty()) {
            TreeNode* node = queue.front();
            queue.pop();
            if (node->left == nullptr && node->right == nullptr && node->val == targetSum){
                return true;
            }
            if (node->left != nullptr) {
                // 直接更新左子节点的值为累计和
                node->left->val += node->val;
                queue.push(node->left);
            }
            if (node->right != nullptr) {
                // 直接更新右子节点的值为累计和
                node->right->val += node->val;  
                queue.push(node->right);
            }
        }
        return false;
    }
};
```
```c++
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        if (root == nullptr)
            return false;
        if (root->left == nullptr && root->right == nullptr) {
            return root->val == targetSum;
        }
        bool left = hasPathSum(root->left, targetSum - root->val);
        bool right = hasPathSum(root->right, targetSum - root->val);

        return left || right;
    }
};
```
